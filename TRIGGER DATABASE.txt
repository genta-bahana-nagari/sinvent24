DELIMITER $$
CREATE TRIGGER barang_masuk
AFTER INSERT ON barang_masuk FOR EACH ROW
begin
  UPDATE barang
  SET barang.stok = barang.stok + NEW.qty_masuk
  WHERE barang.id=NEW.barang_id;
END$$
DELIMITER ;


DELIMITER $$
CREATE TRIGGER barang_keluar
AFTER INSERT ON barang_keluar FOR EACH ROW
begin
  UPDATE barang
  SET barang.stok = barang.stok - NEW.qty_keluar
  WHERE barang.id=NEW.barang_id;
END$$
DELIMITER ;
